var inorderTraversal = function(root) {
    const arr = [];
    function pushRoot(root){
        if (root != null) {
            if (root.left != null) {
                pushRoot(root.left);
            }
            arr.push(root.val);
            if (root.right != null) {
                pushRoot(root.right);
            }
        }
    }
    pushRoot(root)
    return arr
};
