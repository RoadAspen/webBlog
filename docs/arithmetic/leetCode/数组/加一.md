var plusOne = function(digits) {
    let len = digits.length - 1;
    let num = 1;
    while(true){
        if(digits[len] + num === 10){
            digits[len] = 0;
            if(len === 0){
                digits.unshift(1)
                break;
            }
        }else{
            digits[len] += num 
            break;
        }
        len --
    }
    return digits
};
