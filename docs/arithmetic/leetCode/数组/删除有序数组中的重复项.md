var removeDuplicates = function(nums) {
  if (!nums?.length) return;
  if (nums.length === 1) return 1;
  let i = 0;
  for (let j = 1; j < nums.length; j++) {
    if (nums[i] !== nums[j]) {
      nums[++i] = nums[j];
    }
  }
  return i + 1;
};
