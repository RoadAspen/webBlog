## new

使用 new 调用函数

new 操作符实际上执行了以下步骤

```javascript
function _new(fn, ...args)
	//先创建一个对象
	let obj = {};
	obj.__proto__ = fn.prototype;
	let res = fn.apply(obj, args);
	// 返回 res
	return res instanceof Object ? res : obj;
}
```

## call

call 为了改变函数中的 this 指向。思路为将函数赋予新对象的同名属性，这是 this 会自动绑定到新对象上边，该实现为精简版，具体容错处理可以自行扩充,需要直接执行，所以绑定成功之后直接执行。

```js
Function.prototype.mycall = function() {
	// 将传入的参数转换为数组
	let args = [...arguments];
	// 保存函数此时this指向函数本身上下文
	let ctx = this;
	// 创建一个名称，Symbol创建不会重复
	let name = Symbol("name");
	// 获取传入的this指向
	let new_ctx = args[0] || window;
	// 我们只需要把this指向新的对象
	new_ctx[name] = ctx;
	// 将 剩余参数分隔出来
	let residue_args = args.slice(1);
	// 传入剩余参数
	const result = new_ctx[name](...residue_args);
	// 执行成功之后需要删除，否则会在新对象的属性中出现这个方法。
	delete new_ctx[name];

	return result;
};
```

## bind

bind 在于不会立即执行，而会在未来某个时间位置执行。所以需要返回一个函数。

```js
Function.prototype.mybind = function() {
	if (typeof this !== "function") {
		throw new TypeError("this no a function");
	}
	// 将传入的参数转换为数组
	let args = [...arguments];
	// 保存函数此时this指向函数本身
	let ctx = this;
	// 创建一个名称，Symbol创建不会重复
	let name = Symbol("name");
	// 获取传入的this指向
	let new_ctx = args[0] || window;
	// 我们只需要把this指向新的对象
	new_ctx[name] = ctx;
	// 将剩余参数取出来
	let residue_args = args.slice(1);
	let new_func = function() {
		return new_ctx[name](...residue_args);
	};
	// 将原函数的原型复制到新函数上
	new_func.prototype = ctx.prototype;
	return new_func;
};
```

## instanceof

instanceof 用来判断一个构造函数的 prototype 属性所指向的对象是否存在与检测对象的原型链上。

```js
function myInstanceOf(left, right) {
	// left 是检测对象，通过对检测对象原型链查找是否存在 right.prototype
	let proto = left.__proto__;
	let prototype = right.prototype;
	// 使用 while true 代替递归
	while (true) {
		if (proto === null) {
			return false;
		}
		if (proto === prototype) {
			return true;
		}
		proto = proto.__proto__;
	}
}
```

## debounce

防抖 在事件被触发 n 秒后再执行回调，如果在这 n 秒内再次触发，则重新计时。 生活中：如电梯关门，每次进人都会重新等待一段时间关门。

```js
function debounce(fn, wait = 1000) {
	let timer; // 定一个空定时器
	return function() {
		if (timer) {
			clearTimeout(timer);
			timer = null;
		}
		timer = setTimeout(() = {
			fn.apply(this, arguments);
		}, wait);
	};
}

let diantiguanmen = debounce(function() {
	console.log("关门");
}, 10000);

if (有人来了) {
	diantiguanmen();
}
```

## throttle

节流。 事件被触发 n 秒后再执行回调，如果在这 n 秒内再次触发，则不做任何反应，直到 n 秒后再接受新的事件触发指令。
高频事件触发，一段时间内只执行一次

```js
// 利用定时器
function throttle_timer(fn, wait = 1000) {
	let timer;
	return function() {
		if (!timer) {
			timer = setTimeout(() = {
				fn.apply(this, arguments);
			}, wait);
		}
	};
}

// 使用闭包 和时间对比
function throttle_time(fn, wait = 1000) {
	let prev = new Date().getTime();
	return function() {
		let now = new Date().getTime();
		// 如果当前时间过去一定间隔
		if (now - time  wait) {
			fn.apply(this, arguments);
			prev = new Date().getTime();
		}
	};
}
```

## reduce

实现 Array.reduce 函数，累加器，接受一个回调函数和一个初始值（第一次调用 callback 时的传入值），这个回调函数接受四个参数

1. Accumulator (acc) (累计器的值)
2. Current Value (cur) (当前值)
3. Current Index (idx) (当前索引)
4. Source Array (src) (源数组)

```js
Array.prototype._reduce = function(callback, initicalValue) {
	let index = 0;

	if (typeof initicalValue == "undefined") {
		initicalValue = this[0];
		index = 1;
	}
	for (let i = index; i < this.length; i++) {
		acc = callback(initicalValue, this[i], i, this);
	}
	return acc;
};
```

## filter

数组过滤，满足特定条件的项放入新的数组中，callback 接受一个当前项参数,index ,原数组

```js
Array.prototype._filter = function(callback) {
	const new_arr = [];
	if (typeof callback !== "function") return new_arr;
	for (let i = 0; i < this.length; i++) {
		if (callback(this[i], i, this)) {
			new_arr.push(this[i]);
		}
	}
	return new_arr;
};
```

## flat

数组扁平化，将数组嵌套的子数组摊平。返回一个新数组

方法一 使用 reduce 与 concat 替代

```js
function flat(arr) {
	let new_arr = arr.reduce((acc, value) = {
		return acc.concat(Array.isArray(value) ? flat(value) : value);
	}, []);
	return new_arr;
}
```

方法二 使用 递归

```js
let result = [];
function flat(arr) {
	for (let i = 0; i < arr.length; i++) {
		if (Array.isArray(arr[i])) {
			flat(arr[i]);
		} else {
			result.push(arr[i]);
		}
	}
}
```

## jsonStringify

JSON.stringify 方法将对象转换为 json 字符串，由于 undefined ，函数 ，Symbol 格式的数据在 JSON.stringify 中会被置为 `null（数组）`或者 `忽略（对象属性值中出现，key被忽略）`，循环引用会直接报错。正常数字都可以转换为字符串，NaN 和 Infinity 在数组和对象中都会转化为 null

```js
// 主要在于处理 undefined Symbol 和 function 函数， NaN ，Infinity的字符串转换
// new Date() 和 正则会被视为对象，需要单独处理
// 对象中不可枚举属性直接忽略
// 循环引用,直接抛错
function jsonStringify(obj) {
	// 如果循环引用，则直接报错
	try {
		JSON.stringify(obj);
	} catch (e) {
		throw Error(e);
	}
	// 先判断是否是对象
	let type = typeof obj;
	// 如果是非对象
	if (type !== "object" || type === null) {
		return String(obj);
	} else {
		let res;
		let arr = Array.isArray(obj);
		// 如果是 日期 和 正则
		if (obj instanceof Date || obj instanceof RegExp) {
			return String(obj);
		}
		// 数组 或者 对象
		if (arr) {
			// 如果是数组，则对数组内部每一项做递归
			res = obj.map((item) = {
				return jsonStringify(item);
			});
		} else {
			res = [];
			// 如果是对象，则对每一项做递归, 并作为value 存入 数组中,for in 不会遍历 Symbol 类型的属性
			for (key in obj) {
				res.push(key + ":" + jsonStringify(obj[key]));
			}
		}
		// 此时 res 是一个数组,可以直接使用 String 方法转换为没括号的字符串
		return (arr ? "[" : "{") + String(res) + (arr ? "]" : "}");
	}
}
function bs() {
	console.log(Symbol(1));
}
const arr = [
	1,
	2,
	3,
	undefined,
	Symbol(2),
	bs,
	{ a: [1, 2, 3, { a: 666 }] },
	new Date(),
	/^aaa$/,
];
// arr.push(arr)
const a = { name: 1, bs: bs, arr: arr };
console.log(jsonStringify(a));
```

## observer

发布者负责发布事件，订阅者负责订阅事件。

```js
class Observer {
	constructor() {
		this.observerList = {};
	}
	on(type, callback) {
		if (!(callback instanceof Function))
			throw new Error("callback is not a function");
		let index = 0;
		if (this.observerList[type] instanceof Array) {
			index = this.observerList[type].length;
			this.observerList[type].push({
				token: index,
				callback,
			});
		} else {
			this.observerList[type] = [];
			this.observerList[type].push({
				token: 0,
				callback,
			});
		}
		const that = this;
		// 返回一个删除当前监听的函数，无论传入的函数是否是匿名函数
		return function() {
			let len = that.observerList[type].length;
			while (len  0) {
				if (that.observerList[type][len - 1].token === index) {
					that.observerList[type].splice(len - 1);
				}
				len--;
			}
			[index] === undefined;
		};
	}
	emit(type) {
		// 如果 存在，则轮流执行监听
		if (this.observerList[type]) {
			this.observerList[type].forEach((cb) = cb.callback());
		}
	}
	// 移除某个名字下订阅的函数，该方法只能删除命名函数，不能删除匿名函数监听
	removeListener(type, fn) {
		if (this.observerList[type]) {
			this.observerList[type] = this.observerList[type].filter(
				(cb) = cb.callback != fn
			);
		}
	}
	// 只执行一次订阅的事件，然后移除
	once(type, callback) {
		// 绑定的是fn, 执行的时候会触发fn函数
		let fn = () = {
			callback(); // fn函数中调用原有的callback
			this.removeListener(type, fn); // 删除fn, 再次执行的时候之后执行一次
		};
		this.on(type, fn);
	}
}

const observer = new Observer();

observer.on("heihei", function() {
	console.log("heihei");
});

observer.emit("heihei");
observer.emit("heihei");
function ha() {
	console.log("ha");
}
observer.once("ha", ha);

observer.emit("ha");
observer.emit("ha");
```

## curry

柯里化

```js
function curry(fn, args) {
	// 先获取函数的入参个数
	let len = fn.length;
	// args 用来存储每次传入的参数
	args = args || [];
	return function() {
		// 新的数组
		newArgs = args.concat(Array.from(arguments));
		console.log(newArgs);
		if (newArgs.length < len) {
			// 如果参数不够，则再次返回curry函数
			return curry.call(this, fn, newArgs);
		} else {
			// 当参数个数传入完全时，才真正执行
			return fn.apply(this, newArgs);
		}
	};
}

function redu(a, b, c, d) {
	return a + b + c + d;
}

const new_redu = curry(redu);

console.log(new_redu(1, 2, 3, 4));

console.log(new_redu(1)(2, 3)(4));
```

## deepCopy

深拷贝

```js
// 深拷贝主要是基于value为对象或者数组时，浅拷贝会导致引用问题。只能拷贝可枚举属性
function deepCopy(obj) {
	let type = typeof obj;
	if (type !== "object" || obj === null) {
		// 如果值是非数组和对象类型，则直接返回
		return obj;
	} else {
		if (Array.isArray(type)) {
			return obj.map((item = deepCopy(item)));
		} else {
			let new_obj = {};
			for (key in obj) {
				new_obj[key] = deepCopy(obj[key]);
			}
			return new_obj;
		}
	}
}
const cc = { s: 1 };
const a = {
	name: "username",
	age: 11,
	b: cc,
	fun: function() {
		console.log(a);
	},
	arr: [undefined, NaN],
};

const b = deepCopy(a);

console.log(a.b == cc); //true
console.log(b.b == cc); //false
console.log(b.b === a.b); // false
```

## DateFormat

日期解析，将 new Date（）返回的数据按照给定的格式化,主要利用 正则 test 和 replace 之后，会将 括号内规则匹配的赋值给 RegExp $1 $2 ，以此做替换。

```js
Date.prototype.format = function(fmt) {
	let o = {
		"M+": this.getMonth() + 1, // 月份
		"d+": this.getDate(), // 日期
		"h+": this.getHours(), // 小时
		"m+": this.getMinutes(), // 分钟
		"s+": this.getSeconds(), // 秒
		"q+": Math.floor((this.getMonth() + 3) / 3), // 季度
		S: this.getMilliseconds(), // 毫秒
	};
	if (/(y+)/.test(fmt)) {
		fmt = fmt.replace(
			RegExp.$1,
			(this.getFullYear() + "").substr(4 - RegExp.$1.length)
		);
	}

	for (let key in o) {
		if (new RegExp("(" + key + ")").test(fmt)) {
			fmt = fmt.replace(
				RegExp.$1,
				RegExp.$1.length === 1
					? o[key]
					: ("00" + o[key]).substr(("" + o[key]).length)
			);
		}
	}
	return fmt;
};
let time = new Date();
let times = time.format("yyyy-MM-dd hh:mm:ss S");
console.log(time);
console.log(times);
```

## numberParse

```js
function numberParse(number) {
	var n = number;
	var r = "";
	do {
		mod = n % 1000;
		n = n / 1000;
		r = ~~mod + (!!r ? "," + r : "");
	} while (n > 1);

	var strNumber = number + "";
	var index = strNumber.indexOf(".");
	if (index > 0) {
		r += strNumber.substring(index);
	}
	return r;
}
```
